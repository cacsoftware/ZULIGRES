update producto set stock_primera=0, stock_segunda=0, stock_extrusion=0, stock_cargue=0;


select  id_producto, nom_producto, stock_primera, stock_segunda, stock_extrusion, stock_cargue
from producto 
where id_producto in (9, 10, 22)



UPDATE producto SET stock_primera = CASE  
										WHEN id_producto = 10  THEN stock_primera +  60   
									END 
WHERE id_producto IN (10)  

select  id_producto, nom_producto, stock_primera, stock_segunda, stock_extrusion, stock_cargue, 
		pvp_primera as pvp_unid_1ra, pvp_segunda as pvp_unid_2da, (stock_primera * pvp_primera) as total_pvp_1ra, 
		(stock_segunda * pvp_segunda) as total_pvp_2da, (stock_primera * pvp_primera)+(stock_segunda * pvp_segunda) as total_pvp,
		peso * stock_primera as ton_1ra, peso * stock_segunda as ton_2da, (peso * stock_primera)+(peso * stock_segunda) as total_ton
		
from producto 
where id_producto in (9, 10, 22)
